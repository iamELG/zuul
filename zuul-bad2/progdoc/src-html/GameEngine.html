097    /**
098     * Given a command, process (that is: execute) the command.
099     * If this command ends the game, true is returned, otherwise false is
100     * returned.
101     */
102    public void interpretCommand(String commandLine) 
103    {
104        gui.println(commandLine);
105        Command command = parser.getCommand(commandLine);
106
107        if(command.isUnknown()) {
108            gui.println("I don't know what you mean...");
109            return;
110        }
111        
112        String vCommand = command.getCommandWord();
113        switch(vCommand){
114            case "go"  :goRoom(command); break;
115            case "help":printHelp()     ; break;
116            case "look":look()          ; break;
117            case "eat" :eat()           ; break;
118            case "quit":endGame(); break;
119        }
120    }
121

132        gui.print("Your command words are: " + parser.showCommands());
133    }

140    }//printLocationInfo
141
142    /** 
143     * Try to go to one direction. If there is an exit, enter the new
144     * room, otherwise print an error message.
145     */
146    private void goRoom(Command command) 
147    {
148        if(!command.hasSecondWord()) {
149            // if there is no second word, we don't know where to go...
150            gui.println("Go where?");
151            return;
152        }
153
154        String direction = command.getSecondWord();
155
156        // Try to leave current room.
157        Room nextRoom = aCurrentRoom.getExit(direction);
158
159        if (nextRoom == null)
160            gui.println("There is no door!");
161        else {
162            aCurrentRoom = nextRoom;
163            gui.println(aCurrentRoom.getLongDescription());
164            if(aCurrentRoom.getImageName() != null)
165                gui.showImage(aCurrentRoom.getImageName());
166        }
167    }
168

172        gui.enable(false);
173    }
