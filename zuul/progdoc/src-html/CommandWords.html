001import java.util.HashMap;
002/**
003 * This class is part of "Beewick castle" application. 
004 * "Beewick castle" is a very simple, text based adventure game.  
005 * 
006 * This class holds an enumeration table of all command words known to the game.
007 * It is used to recognise commands as they are typed in.
008 *
009 * @author  Michael Kolling and David J. Barnes + D.Bureau & Elvin Guilloton
010 * @version 2008.03.30 + 2013.09.15
011 */
012public class CommandWords
013{
014    // tableau constant qui contient tous les mots de commande valides
015    private HashMap<String, CommandWord> validCommands;
016
017    /**
018     * Constructeur par defaut
019     */
020    public CommandWords(){
021        validCommands = new HashMap<String, CommandWord>();
022        for(CommandWord command : CommandWord.values()) {
023            if(command != CommandWord.UNKNOWN) {
024                validCommands.put(command.toString(), command);
025            }
026        } 
027    } // CommandWords()
028
029    /**
030     * Check whether a given String is a valid command word. 
031     * @return true if it is, false if it isn't.
032     */
033    public boolean isCommand(String aString){
034        return validCommands.containsKey(aString);
035    }
036    
037    /**
038     * return all valid commands
039     */
040    public String getCommandList() {
041        String vString="";
042        for(String command : validCommands.keySet()) {
043            vString+=command + "  ";
044        }        
045        return vString;
046    }
047    
048    /**
049     * Find the CommandWord associated with a command word.
050     * @param commandWord The word to look up.
051     * @return The CommandWord correspondng to commandWord, or UNKNOWN
052     *         if it is not a valid command word.
053     */
054    public CommandWord getCommandWord(String commandWord)
055    {
056        CommandWord command = validCommands.get(commandWord);
057        if(command != null) {
058            return command;
059        }
060        return CommandWord.UNKNOWN;
061    }
062} // CommandWords
