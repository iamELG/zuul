013public class Room{
014    private String aDescription;
015    private HashMap<String,Room> aExits;
016    private HashMap<String,Door> aDoor;
017    private String aImageName;
018    private ItemList aItem;
019
020    /**
021     * natural constructor of the class Room
022     * @param pDescription description of the room
023     * @param pImage the destination to the image file of the room
024     * @param pItem a item that will be n the room
025     */
026    public Room(final String pDescription, final String pImage,Item pItem){   
027        this.aDescription=pDescription;
028        this.aImageName=pImage;
029        aExits=new HashMap<String, Room>();
030        aDoor=new HashMap<String, Door>();
031        aItem=new ItemList();
032        if(pItem!=null)
033            aItem.addItem(pItem);
034    }//Room 
035    
036    /**
037     * constructor of the class Room without any object
038     * @param pDescription description of the room
039     * @param pImage the destination to the image file of the room
040     */
041    public Room(final String pDescription, final String pImage){   
042        this(pDescription,pImage,null);
043    }//Room 
044    
045    /** 
046     * add an Exit to the HashMap aExits
047     * @param pDir the direction that lead to the next room
048     * @param pRoom the Room you want in the direction
049     */
050    public void setExits(String pDir,Room pRoom){
051        aExits.put(pDir,pRoom);
052    }//setExits
053   
054    /** 
055     * add a Door to the HashMap aDoor
056     * @param pDir the direction  in wich the Door is 
057     * @param pDoor the Door you want in the direction
058     */
059    public void setDoor(String pDir,Door pDoor){
060        aDoor.put(pDir,pDoor);
061    }//setExits
062    
063    /**
064     * return the attribute aDescription
065     * @return aDescription
066     */
067    public String getDescription(){
068        return(this.aDescription);
069    }//getDescription
070    
071    /**
072     * return a Room for the HashMap aExit
073     * @param pDirection the direction you want the Exit of. 
074     * @return the Room 
075     */
076    public Room getExit(String pDirection){
077        return(aExits.get(pDirection));
078    }//getExit
079    
080    /**
081     * return a Door for the HashMap aDoor
082     * @param pDirection the direction you want the Door of. 
083     * @return the Door 
084     */
085    public Door getDoor(String pDirection){
086        return(aDoor.get(pDirection));
087    }//getExit
088        
089    /**
090     * return the attribute ItemList
091     * @return the attribute aItem
092     */
093    public ItemList getItemList(){
094        return aItem;
095    }//getItemList
096    
097    /**
098     * return the String of all the Exits with a Space inbetween 
099     * @return String of all the Exits
100     */
101    private String getExitString()
102    {
103        StringBuilder vreturnString = new StringBuilder( "Exits:" );
104        for ( String vS : aExits.keySet() )
105            vreturnString.append( " " + vS );
106        return vreturnString.toString();
107    }//getExitString
108    
109    /**
110     * return a String with the long description of the room
111     * @return a String with the description of the room
112     */
113    public String getLongDescription(){
114        String vStr="You are in the: " + aDescription+".\n"+getExitString();
115        
116        if(aItem.isEmpty())
117            return (vStr);
118           
119        return (vStr+"\nItem:"+aItem.getItemString());
120    }//getLongDescription
121
122    /**
123     * return the name of the Image
124     * @return the String of a Image file
125     */
126    public String getImageName(){
127        return aImageName;
128    }//getImageName
