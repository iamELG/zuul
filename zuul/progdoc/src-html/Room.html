001import java.util.HashMap;
002import java.util.HashMap;
003import java.util.Set;
004/**
005 * This class is part of "Beewick castle" application. 
006 * "Beewick castle" is a very simple, text based adventure game.  
007 * 
008 * This class is to move move around the map in diferent room.
009 * 
010 * @author  Michael Kolling and David J. Barnes + D.Bureau & Elvin Guilloton
011 * @version 2008.03.30 + 2013.09.15
012 */
013public class Room
014{
015    private String aDescription;
016    private HashMap<String,Room> aExits;
017    private String aImageName;
018    private ItemList aItem;
019
020    /** constructeur naturel
021     * 
022     */
023    public Room(final String pDescription, final String pImage,Item pItem)
024    {   
025        this.aDescription=pDescription;
026        this.aImageName=pImage;
027        aExits=new HashMap<String, Room>();
028        aItem=new ItemList(); 
029        aItem.addItem(pItem);
030    }//Room 
031    
032    /** constructeur sans objet
033     * 
034     */
035    public Room(final String pDescription, final String pImage)
036    {   
037        this.aDescription=pDescription;
038        this.aImageName=pImage;
039        aExits=new HashMap<String, Room>();
040        aItem=new ItemList();
041    }//Room 
042    
043    /** setExits cree les sortie dans l'atribut HashMap
044     *
045     */
046    public void setExits(String pDir,Room pNei){
047        aExits.put(pDir,pNei);
048    }//setExits
049   
050    /** accesseur
051     * 
052     */
053    public String getDescription(){
054        return(this.aDescription);
055    }//getDescription
056    
057    /** acceseur
058     *
059     */
060    public Room getExit(String direction){
061        return(aExits.get(direction));
062    }//getExit
063    
064    /**geter
065     * 
066     */
067    public ItemList getItem(){
068        return aItem;
069    }
070    
071    /** getExitString acceseur des sortie
072     *
073     */
074    private String getExitString()
075    {
076        StringBuilder vreturnString = new StringBuilder( "Exits:" );
077        for ( String vS : aExits.keySet() )
078            vreturnString.append( " " + vS );
079        return vreturnString.toString();
080    }//getExitString
081    
082    /** return a lomg description of this room
083     *
084     */
085    public String getLongDescription(){
086        String vStr="You are in the: " + aDescription+".\n"+getExitString();
087        
088        if(aItem.isEmpty())
089            return (vStr);
090           
091        return (vStr+"\n"+aItem.getItemString());
092    }//getLongDescription
093
094    /**
095     * Return a string describing the room's image name
096     */
097    public String getImageName(){
098        return aImageName;
099    }//getImageName
100    
101} // Room
