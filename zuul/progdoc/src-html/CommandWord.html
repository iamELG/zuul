001/**
002 * This class is part of "Beewick castle" application. 
003 * "Beewick castle" is a very simple, text based adventure game.
004 * Representations for all the valid command words for the game.
005 * 
006 * @author Michael Kolling and David J. Barnes & Elvin Guiloton
007 * @version (Jan 2018)
008 */
009public enum CommandWord{
010    // A value for each command word, plus one for unrecognised
011    // commands.
012    GO("go"), QUIT("quit"), HELP("help"), LOOK("look"), EAT("eat"), BACK("back"), TEST("test"), TAKE("take"), DROP("drop"), ITEMS("items"),CHARGE("charge"),FIRE("fire"), UNKNOWN("?");
013    // The command string.
014    private String commandString;
015    
016    /**
017     * Initialise with the corresponding command word.
018     * @param commandWord The command string.
019     */
020    CommandWord(String commandString){
021        this.commandString = commandString;
022    }//CommandWord
023    
024    /**
025     * @return The command word as a string.
026     */
027    public String toString(){
028        return commandString;
029    }//toString
030}//CommandWord
