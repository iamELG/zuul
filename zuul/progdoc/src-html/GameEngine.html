111    /**
112     * Given a command, process (that is: execute) the command.
113     * If this command ends the game, true is returned, otherwise false is
114     * returned.
115     */
116    public void interpretCommand(String commandLine){
117        gui.println(commandLine);
118        Command vCommand = parser.getCommand(commandLine);
119
120        if(vCommand.isUnknown()) {
121            gui.println("I don't know what you mean...");
122            return;
123        }
124        
125        String vCommandSTR = vCommand.getCommandWord();
126        switch(vCommandSTR){
127            case "go"  :goRoom(vCommand); break;
128            case "help":printHelp()     ; break;
129            case "look":look()          ; break;
130            case "eat" :eat()           ; break;
131            case "back":back()          ; break;
132            case "test":test(vCommand)  ; break;
133            case "take":take(vCommand)  ; break;
134            case "drop":drop(vCommand)  ; break;
135            case "quit":endGame(); break;
136        }
137    }

158    /** 
159     * Try to go to one direction. If there is an exit, enter the new
160     * room, otherwise print an error message.
161     */
162    private void goRoom(Command pCommand){
163        if(!pCommand.hasSecondWord()) {
164            // if there is no second word, we don't know where to go...
165            gui.println("Go where?");
166            return;
167        }
168
169        String vDirection = pCommand.getSecondWord();
170
171        // Try to leave current room.
172        Room nextRoom = aCurrentRoom.getExit(vDirection);
173
174        if (nextRoom == null)
175            gui.println("There is no door!");
176        else {
177            aStack.push(aCurrentRoom);
178            aCurrentRoom = nextRoom;
179            gui.println(aCurrentRoom.getLongDescription());
180            if(aCurrentRoom.getImageName() != null)
181                gui.showImage(aCurrentRoom.getImageName());
182        }
183    }
184
185    /** permet de retourner dans la room precedente
186     * 
187     */
188    private void back(){
189        if(aStack.empty()){
190            gui.println("you are already at the beginning!");
191            gui.println("what did you expect? go back in time?");
192            return;
193        }
194        Room vRoom=aStack.pop();
195        aCurrentRoom = vRoom;
196        gui.println(aCurrentRoom.getLongDescription());
197        if(aCurrentRoom.getImageName() != null)
198            gui.showImage(aCurrentRoom.getImageName());
199    }

224    /**test

226     * 
227     */
228    private void test(Command pCommand){
229        if(!pCommand.hasSecondWord()){
230            gui.println("you didn't specify the file");
231            return;
232        }  
233        Scanner vSc;
234        try { // pour "essayer" les instructions suivantes
235            String pNomFichier= pCommand.getSecondWord();
236            vSc = new Scanner( new File( pNomFichier ) );
237            while ( vSc.hasNextLine() ) {
238                String vLigne = vSc.nextLine();
239                interpretCommand(vLigne);
240                // traitement de la ligne lue
241            } // while
242        } // try
243        catch ( final FileNotFoundException pFNFE ) {
244            gui.println("no such file or directory:"+pCommand.getSecondWord());
245        } // catch
246    }//teste
247    /**
248     *  take
249     */
250    private void take(Command pCommand){
251        if(!pCommand.hasSecondWord()) {
252            gui.println("which object do you want too take?");
253            return;
254        }
255        String vName = pCommand.getSecondWord();
256        
257        if(!aCurrentRoom.getItem().itemInList(vName)){
258            aPlayer.getItem().addItem(aCurrentRoom.getItem().getItem(vName));
259            aCurrentRoom.getItem().removeItem(vName);
260            look();
261            return;
262        }
263        gui.println("this item is not in the room!");
264        look();
265    }//take
