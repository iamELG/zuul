135    /**
136     * Given a command, process (that is: execute) the command.
137     * If this command ends the game, true is returned, otherwise false is
138     * returned.
139     */
140    public boolean interpretCommand(Command pCommand){
141        
142        CommandWord vCommandWord = pCommand.getCommandWord();
143        
144        if(vCommandWord== CommandWord.UNKNOWN) {
145            gui.println("I don't know what you mean...");
146            return false;
147        }
148        
149        switch(vCommandWord){
150            case GO  :goRoom(pCommand); return true;
151            case EAT :eat(pCommand)   ; return true;
152            case TEST:test(pCommand)  ; return true;
153            case TAKE:take(pCommand)  ; return true;
154            case DROP:drop(pCommand)  ; return true;
155            case HELP:printHelp()     ; return true;
156            case LOOK:look()          ; return true;
157            case BACK:back()          ; return true;
158            case ITEMS:items()        ; return true;
159            case CHARGE:charge()      ; return true;
160            case FIRE:fire()          ; return true;
161            case QUIT:endGame()       ; return true;
162        }
163        return false;       
164    }//interpretCommand

194    /** 
195     * Try to go to one direction. If there is an exit, enter the new
196     * room, otherwise print an error message.
197     */
198    private void goRoom(Command pCommand){
199        gui.println("");
200        if(!pCommand.hasSecondWord()) {
201            // if there is no second word, we don't know where to go...
202            gui.println("Go where?");
203            return;
204        }
205
206        String vDirection = pCommand.getSecondWord();
207
208        // Try to leave current room.
209        Room nextRoom = aPlayer.getCurrentRoom().getExit(vDirection);
210
211        if (nextRoom == null){
212            gui.println("There is no door!");
213            return ;
214        }
215        boolean pStack=true;
216        if(aPlayer.getCurrentRoom().getDoor(vDirection)!=null){
217            if(aPlayer.getCurrentRoom().getDoor(vDirection).isTrapDoor()){
218                if(!aPlayer.getCurrentRoom().getDoor(vDirection).canGo()){
219                    gui.println("you can go through this door\nat least not in this direction!");
220                    return;
221                }
222                if(aPlayer.getCurrentRoom().getDoor(vDirection).canGo()){
223                    pStack=false;
224                    aPlayer.getStack().clear();
225                }            
226            }
227            else if(aPlayer.getCurrentRoom().getDoor(vDirection).isLocked()){
228                if(!aPlayer.getInventory().itemInList("key")){
229                    gui.println("this door is locked, but you open it with your key");
230                    aPlayer.getCurrentRoom().getDoor(vDirection).setLocked(false);
231                }
232                else{
233                    gui.println("this door seems to be locked, you will need a key");
234                    return;
235                }
236            }
237        }             
238        if(pStack)
239            aPlayer.getStack().push(aPlayer.getCurrentRoom());
240            
241        if(aPlayer.getCurrentRoom().isTransportRoom())
242            aPlayer.getStack().clear();
243            
244        aPlayer.setCurrentRoom(nextRoom);
245        gui.println(aPlayer.getCurrentRoom().getLongDescription());
246        if(aPlayer.getCurrentRoom().getImageName() != null)
247            gui.showImage(aPlayer.getCurrentRoom().getImageName());   
248        win();
249        addAMove();
250    }//goRoom

299    /**
300     * eat let you item edible item, edible item double you maximum weight you can carry
301     */
302    private void eat(Command pCommand){
303        if(!pCommand.hasSecondWord()){
304            gui.println("what do you want to eat?");
305            return;
306        }
307        String vName = pCommand.getSecondWord();
308        if(aPlayer.getInventory().itemInList(vName)){
309            gui.println("you dont have any "+vName);
310            return;
311        }
312        if(aPlayer.getInventory().getItem(vName).getEdible()){
313            gui.println("this item is not food, you can't eat it");
314            return;
315        }
316        
317        aPlayer.setMaxWeight(aPlayer.getMaxWeight()*2);
318        aPlayer.getInventory().removeItem(vName);
319        gui.println("you ate a cookie, you can now carry more loot");
320    }//eat

372    /**
373     *  take an item put it from the room to the player inventory
374     */
375    private void take(Command pCommand){
376        if(!pCommand.hasSecondWord()) {
377            gui.println("which object do you want too take?");
378            return;
379        }
380        String vName = pCommand.getSecondWord();
381        
382        if(aPlayer.getCurrentRoom().getItemList().itemInList(vName)){
383            gui.println("this item is not in the room!");
384            look();   
385            return;
386        }
387        
388        if(aPlayer.getCurrentWeight()+aPlayer.getCurrentRoom().getItemList().getItem(vName).getWeight() > aPlayer.getMaxWeight()){
389            gui.println("this item is too heavy for you!");
390            look();   
391            return;
392        }
393        aPlayer.setCurrentWeight( aPlayer.getCurrentWeight() + aPlayer.getCurrentRoom().getItemList().getItem(vName).getWeight() );
394        aPlayer.getInventory().addItem(aPlayer.getCurrentRoom().getItemList().getItem(vName));
395        aPlayer.getCurrentRoom().getItemList().removeItem(vName);
396        look();    
397    }//take
