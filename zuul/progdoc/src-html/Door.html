002/**
003 * This class is part of "Beewick castle" application. 
004 * "Beewick castle" is a very simple, text based adventure game.
005 * this classe is to stop Player from mooving with some locked door/ trap door.
006 * @author Elvin Guilloton
007 * @version 1.0 (Jan 2018)
008 */
009public class Door{
010    private boolean aLocked;
011    private boolean aIsTrapDoor;
012    private boolean aGoodDirection;
013    
014    /** constructeur naturel
015     * 
016     */
017    public Door(boolean pLocked,boolean pIsTrapDoor,boolean pGoodDirection){
018        aGoodDirection=pGoodDirection;
019        aIsTrapDoor=pIsTrapDoor;
020        aLocked=pLocked;
021    }
022    
023    public boolean isTrapDoor(){
024        return aIsTrapDoor;
025    }
026    
027    public boolean isLocked(){
028        return aLocked;        
029    }
030    
031    public boolean canGo(){
032        return aGoodDirection;
033    }
034    
035    public void setLocked(boolean pLocked){
036        aLocked=pLocked;
037    }
038    
039    public void setCanGo(boolean pGoodDirection){
040         aGoodDirection=pGoodDirection;
041    }
042}
